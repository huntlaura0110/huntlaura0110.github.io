              function countBs(bString) {
                var count = 0;
                for (var i = 0; i < bString.length; i++) {
                  if (bString.charAt(i) == 'B') {
                    count = (count + 1);
                  }
                }
                return count;
              }
              $("#q4button").on("click", function() {
                var userInput = prompt("");
                var x = countBs(userInput);
                $("#q4result").html(x);
              });
          
 
              function countChar(cString, charMatch) {

        function sum(array) {
        var sm = 0;
        for (var i = 0; i < array.length; i++) {
        sm += parseInt(array[i]);
        }
        return sm;
        }

                        var sarr = [valueA, valueB, valueC, valueD, valueE];

                        temp = sarr[0]; // store valueA in temp variable
                        sarr[0] = sarr[4]; // valueA takes the value of valueE
                        sarr[4] = temp; //valueE takes the value of temp, which is the stored value of valueA
                        //valueA and valueE have been swapped

                        temp = sarr[1]; //valueB is stored in temp variable
                        sarr[1] = sarr[3]; //valueB takes the value of valueD
                        sarr[3] = temp; //valueD takes the new value of temp, which is the stored value of valueB
                        //valueB and valueD have been swapped

                        //valueC is uneffected because it is the middle value in the array
